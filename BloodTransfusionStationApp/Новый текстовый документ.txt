
CREATE FUNCTION [dbo].[FullTakingBloodInf]()
RETURNS TABLE 
AS
RETURN 
(
	SELECT        Доноры.Имя, Доноры.Фамилия, [Прием крови].[Дата приема], Доноры.[Группа крови], [Прием крови].[Вид донорства], [Прием крови].Объем, Врачи.Фамилия AS [Фамилия врача]
FROM            Врачи INNER JOIN
                         [Прием крови] ON Врачи.Id = [Прием крови].Врач INNER JOIN
                         Доноры ON [Прием крови].Донор = Доноры.[Номер донора]
)
